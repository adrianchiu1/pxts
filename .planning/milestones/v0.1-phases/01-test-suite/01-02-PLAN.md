```python
from pxts.exceptions import pxtsValidationError

def validate_ts(df: pd.DataFrame) -> pd.DataFrame:
    """Raises pxtsValidationError (TypeError subclass) if not DatetimeIndex. Returns df."""

def set_tz(df: pd.DataFrame, tz: str = "UTC") -> pd.DataFrame:
    """
    - Naive index: localize to tz silently (no warning)
    - Already at target tz: return df unchanged (same object, no warning)
    - Tz-aware but different tz: convert with UserWarning
    Returns new DataFrame. Does not modify in place.
    """

def to_dense(df: pd.DataFrame, freq: str, fill: str | None = None) -> pd.DataFrame:
    """
    - Returns df.asfreq(freq, method=fill)
    - No-op guard: if df.index.freq is not None and df.index.freqstr == freq, return df unchanged
    - fill='ffill' or fill='bfill' forward/back fills new rows
    Known edge: no-op uses raw freqstr comparison, so '1D' != 'D' even if equivalent
    """

def infer_freq(df: pd.DataFrame) -> str:
    """
    - Returns pandas offset alias from min(index.diff().dropna())
    - Raises ValueError if fewer than 2 data points
    - Returns 'D' for daily data (cannot distinguish from 'B')
    """
```

    - Valid DatetimeIndex df → returns df unchanged

    - pxtsValidationError is catchable as TypeError (isinstance check)
    - Error message contains "DatetimeIndex"

    **set_tz:**
    - Naive index + tz="UTC" → localized, no warning emitted
    - Already UTC index + tz="UTC" → same object returned, no warning
    - US/Eastern index + tz="UTC" → converted, UserWarning emitted containing "converted"
    - Naive index + tz="US/Eastern" → localized to US/Eastern, no warning
    - Non-DatetimeIndex df → raises pxtsValidationError (validate_ts gate)

    **to_dense:**
    - Sparse daily df (gaps) + freq="D" → densified, NaN in new rows
    - Sparse daily df (gaps) + freq="D" + fill="ffill" → gaps filled with prior value
    - Already-dense df with freq set + matching freqstr → no-op, same object returned
    - Non-DatetimeIndex df → raises pxtsValidationError
    - freq="1D" on df with freqstr "D" → NOT a no-op (known edge case — freqstr mismatch)

    **infer_freq:**
    - 5-row daily df → returns "D"
    - 5-row hourly df (1h apart) → returns "h"
    - Single-row df (1 row) → raises ValueError
    - Empty df → raises ValueError
    - Non-DatetimeIndex df → raises pxtsValidationError
