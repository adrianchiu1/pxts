```python
# accessor.py — registered as @pd.api.extensions.register_dataframe_accessor("ts")
class TsAccessor:
    def __init__(self, pandas_obj: pd.DataFrame) -> None:
        # Raises AttributeError if not DatetimeIndex (pandas protocol requirement)
        if not isinstance(pandas_obj.index, pd.DatetimeIndex):
            raise AttributeError(...)
        self._obj = pandas_obj

    def set_tz(self, tz="UTC") -> pd.DataFrame:         # delegates to core.set_tz
    def to_dense(self, freq, fill=None) -> pd.DataFrame: # delegates to core.to_dense
    def infer_freq(self) -> str:                         # delegates to core.infer_freq
    def write_ts(self, path, *, date_format=None) -> None: # delegates to io.write_ts
    def plot(self, cols=None, ...) -> Figure:            # delegates to plots.tsplot
    def plot_dual(self, left, right, ...) -> Figure:     # delegates to plots.tsplot_dual
```

**Accessor registration side effect:**
Importing `pxts.accessor` (or `import pxts`) registers the `.ts` accessor globally
via `@register_dataframe_accessor("ts")`. This is idempotent — re-registering prints
a warning but doesn't break anything. Import at module level in test file.

  <behavior>
    **Accessor availability:**
    - After `import pxts.accessor`, a valid `ts_df` has a `.ts` attribute
    - `df.ts` is a `TsAccessor` instance

    **Delegation (each method returns the same result as standalone function):**
    - `ts_df.ts.set_tz("UTC")` returns same as `set_tz(ts_df, "UTC")` — types match
    - `ts_df.ts.to_dense("D")` returns a DataFrame with DatetimeIndex
    - `ts_df.ts.infer_freq()` returns a string (the offset alias)
    - `ts_df.ts.write_ts(tmp_path / "out.csv")` creates a file at that path
    - `ts_df.ts.plot(backend="matplotlib")` returns a Figure (close after test)
    - `ts_df.ts.plot_dual(left=["A"], right=["B"], backend="matplotlib")` returns a Figure

    **Error protocol (pandas accessor contract):**
    - `bad_df.ts` raises `AttributeError` (NOT `pxtsValidationError`)
    - The AttributeError message contains "DatetimeIndex"
  </behavior>
  <implementation>
    ```python
    import matplotlib
    matplotlib.use("Agg")
    import matplotlib.pyplot as plt
    import pandas as pd
    import pytest
    import pxts.accessor  # Side effect: registers .ts accessor
    from pxts.core import set_tz

    def test_accessor_available(ts_df):
        from pxts.accessor import TsAccessor
        assert isinstance(ts_df.ts, TsAccessor)

    def test_set_tz_delegates(ts_df):
        result = ts_df.ts.set_tz("UTC")
        assert isinstance(result, pd.DataFrame)
        assert result.index.tz is not None

    def test_bad_df_raises_attribute_error(bad_df):

            _ = bad_df.ts
    ```

<feature>
  <name>test_theme.py — apply_theme registration and isolation tests</name>
  <files>tests/test_theme.py</files>
  <behavior>
    **Plotly template registration:**
    - After `apply_theme()`, `pio.templates["pxts"]` exists (no KeyError)
    - After `apply_theme()`, `pio.templates.default == "pxts"`

    **Matplotlib rcParams mutation:**
    - After `apply_theme()`, `plt.rcParams["axes.facecolor"] == "#FFFFFF"`
    - After `apply_theme()`, `plt.rcParams["axes.grid"] == True`

    **Idempotency:**
    - Calling `apply_theme()` twice in a row does not raise and produces the same state

    **Isolation (critical for TEST-06):**
    - `apply_theme()` must not be called at module import in the test file itself
    - Test must call `apply_theme()` explicitly, then verify, then restore rcParams
    - Use `matplotlib.rcdefaults()` in a teardown/fixture to restore defaults after
      theme tests, preventing contamination of subsequent test files

    Strategy for isolation: use a pytest fixture with `yield` to restore rcParams:
    ```python
    @pytest.fixture(autouse=False)
    def restore_rcparams():
        import copy
        saved = copy.deepcopy(plt.rcParams.copy())
        yield
        plt.rcParams.update(saved)
    ```
  </behavior>
  <implementation>
    ```python
    import matplotlib
    matplotlib.use("Agg")
    import matplotlib.pyplot as plt
    import plotly.io as pio
    import copy
    import pytest
    from pxts.theme import apply_theme

    @pytest.fixture
    def restore_rcparams():
        saved = dict(plt.rcParams)
        yield
        plt.rcParams.update(plt.rcParamsDefault)  # full reset
        plt.rcParams.update(saved)

    def test_plotly_template_registered(restore_rcparams):
        apply_theme()
        assert "pxts" in pio.templates
        assert pio.templates.default == "pxts"

    def test_matplotlib_rcparams_set(restore_rcparams):
        apply_theme()
        assert plt.rcParams["axes.facecolor"] == "#FFFFFF"
        assert plt.rcParams["axes.grid"] == True

    def test_apply_theme_idempotent(restore_rcparams):
        apply_theme()
        apply_theme()  # Must not raise
        assert "pxts" in pio.templates
    ```

    Note: `pxts` calls `apply_theme()` at import time in `__init__.py`. The test file
    imports `from pxts.theme import apply_theme` directly (NOT `import pxts`) to avoid
    triggering the import-time side effect before the restore fixture is active.
  </implementation>
</feature>
