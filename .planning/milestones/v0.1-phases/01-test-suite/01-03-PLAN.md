def _detect_date_format(sample: str) -> tuple[str, bool]:
    """
    Returns (format_string, dayfirst).

    - Slash, first > 12: DD/MM/YYYY  → ("%d/%m/%Y", True)
    - Slash, second > 12: MM/DD/YYYY → ("%m/%d/%Y", False)
    - Slash, ambiguous (both ≤ 12)   → ("%m/%d/%Y", False)   # US default, NO warning (Phase 2 will add warning)
    - Fallback                       → ("mixed", False)
    """
