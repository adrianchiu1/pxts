```python
"""Unit tests for pxts.io.read_bdh — Bloomberg BDH historical data fetcher.

pdblp is mocked throughout so no Bloomberg terminal is required.
"""
from __future__ import annotations

import sys
import pandas as pd
import pytest
from unittest.mock import MagicMock, patch, call

from pxts.io import read_bdh


def _make_bdh_response(tickers: list[str], field: str, dates: list[str]) -> pd.DataFrame:
    """Build a fake pdblp BCon.bdh() MultiIndex response."""
    index = pd.to_datetime(dates)
    import numpy as np
    data = {(t, field): [float(i + j) for j, _ in enumerate(dates)]
            for i, t in enumerate(tickers)}
    columns = pd.MultiIndex.from_tuples(data.keys())
    return pd.DataFrame(list(data.values()), index=columns, columns=index).T
```

For each test, use `@patch('pxts.io.pdblp')` to inject a mock module. The mock module's BCon() call returns a mock connection object with .start(), .bdh(), .stop() methods.

Key mock setup pattern:
```python
@patch('pxts.io.pdblp')
def test_happy_path_single_ticker(self, mock_pdblp):
    mock_con = MagicMock()
    mock_pdblp.BCon.return_value = mock_con
    mock_con.bdh.return_value = _make_bdh_response(
        ['AAPL US Equity'], 'PX_LAST', ['2024-01-02', '2024-01-03']
    )
    df = read_bdh(['AAPL US Equity'], '20240101')
    assert list(df.columns) == ['AAPL US Equity']
    assert isinstance(df.index, pd.DatetimeIndex)
```

For test_importerror_without_pdblp, temporarily remove pdblp from sys.modules and patch the import inside the function. Since read_bdh does `import pdblp` lazily, use:

```python
def test_importerror_without_pdblp(self):
    with patch.dict(sys.modules, {'pdblp': None}):
        with pytest.raises(ImportError, match='pip install pxts\\[bloomberg\\]'):
            read_bdh(['AAPL US Equity'], '20240101')
```
